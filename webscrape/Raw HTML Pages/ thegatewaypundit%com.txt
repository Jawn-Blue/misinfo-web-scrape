			function tgpStickySbScrollListener() {
				tgp_last_known_scroll_position = window.scrollY;

					window.requestAnimationFrame(function() {
						tgpPreventWidgetOverlap(tgp_last_known_scroll_position);
						tgp_sticky_sb_ticking = false;
					})

					tgp_sticky_sb_ticking = true;

			function tgpPreventWidgetOverlap(window_scroll_position) {
				let tgp_sticky_sb_body_height = document.body.offsetHeight;
				let tgp_sticky_sb_widget_height = document.querySelector(".fixed-widgets").offsetHeight;
				let tgp_sticky_sb_footer_height = document.getElementById("tgp-footer").offsetHeight;
				let tgp_sticky_sb_menu_height = document.getElementById("wj-network-header-container").offsetHeight;
				let tgp_sticky_sb_cnx_height = document.querySelector(".cnx-main-container.cnx-in-mobile") ? document.querySelector(".cnx-main-container.cnx-in-mobile").offsetHeight : 0;
				let tgp_sticky_sb_stop_height = tgp_sticky_sb_body_height - tgp_sticky_sb_menu_height - tgp_sticky_sb_widget_height - tgp_sticky_sb_footer_height - tgp_sticky_sb_cnx_height - 10;
				let tgp_sticky_sb_archive_loop = document.getElementById("tgp-archive-loop");

				// once to the footer attatch sticky to bottom instead of top
				if ( ( window_scroll_position > tgp_sticky_sb_stop_height ) && ( document.getElementsByClassName("fixed-widgets fixed").length >= 1 ) ) {
					document.querySelector(".fixed-widgets").classList.add("stop");
				} else {
					document.querySelector(".fixed-widgets").classList.remove("stop");
				}

				// removing fixed-widget-anchor for when content is shorter than asside... otherwise cant scroll to bottom.
				if ( tgp_sticky_sb_archive_loop ) {
					// for archives... "main.content" sometimes doesnt have a height for archives
					if ( tgp_sticky_sb_archive_loop.getBoundingClientRect().bottom < document.querySelector("section.fixed-widgets").getBoundingClientRect().bottom ) {
						document.getElementById("fixed-widget-anchor").remove();
					}
				}else if ( document.querySelector("main.content").getBoundingClientRect().bottom < document.querySelector("section.fixed-widgets").getBoundingClientRect().bottom && document.getElementById("fixed-widget-anchor") ) {
					// for non-archive pages... they tend to have a height
					document.getElementById("fixed-widget-anchor").remove();
				}
			}
