function wpcom_remote_login_remove_dom_node_id( element_id ) {
	var dom_node = document.getElementById( element_id );
	if ( dom_node ) { dom_node.parentNode.removeChild( dom_node ); }
}
function wpcom_remote_login_remove_dom_node_classes( class_name ) {
	var dom_nodes = document.querySelectorAll( '.' + class_name );
	for ( var i = 0; i < dom_nodes.length; i++ ) {
		dom_nodes[ i ].parentNode.removeChild( dom_nodes[ i ] );
	}
}
function wpcom_remote_login_final_cleanup() {
	wpcom_remote_login_remove_dom_node_classes( "wpcom_remote_login_msg" );
	wpcom_remote_login_remove_dom_node_id( "wpcom_remote_login_key" );
	wpcom_remote_login_remove_dom_node_id( "wpcom_remote_login_validate" );
	wpcom_remote_login_remove_dom_node_id( "wpcom_remote_login_js" );
	wpcom_remote_login_remove_dom_node_id( "wpcom_request_access_iframe" );

// Watch for messages back from the remote login
window.addEventListener( "message", function( e ) {
	if ( e.origin === "https://r-login.wordpress.com" ) {
		var data = {};
		try {
			data = JSON.parse( e.data );
		} catch( e ) {
			wpcom_remote_login_final_cleanup();
			return;
		}

		if ( data.msg === 'LOGIN' ) {
			// Clean up the login check iframe
			wpcom_remote_login_remove_dom_node_id( "wpcom_remote_login_key" );

			var id_regex = new RegExp( /^[0-9]+$/ );
			var token_regex = new RegExp( /^.*|.*|.*$/ );
			if (
				token_regex.test( data.token )
				&& id_regex.test( data.wpcomid )
			) {
				// We have everything we need to ask for a login
				var script = document.createElement( "script" );
				script.setAttribute( "id", "wpcom_remote_login_validate" );
				script.src = '/remote-login.php?wpcom_remote_login=validate'
					+ '&wpcomid=' + data.wpcomid
					+ '&token=' + encodeURIComponent( data.token )
					+ '&host=' + window.location.protocol
						+ '//' + window.location.hostname;
				document.body.appendChild( script );
			}

			return;
		}

		// Safari ITP, not logged in, so redirect
		if ( data.msg === 'LOGIN-REDIRECT' ) {
			window.location = 'https://wordpress.com/log-in?redirect_to=' + window.location.href;
			return;
		}

		// Safari ITP, storage access failed, remove the request
		if ( data.msg === 'LOGIN-REMOVE' ) {
			var css_zap = 'html { -webkit-transition: margin-top 1s; transition: margin-top 1s; } /* 9001 */ html { margin-top: 0 !important; } * html body { margin-top: 0 !important; } @media screen and ( max-width: 782px ) { html { margin-top: 0 !important; } * html body { margin-top: 0 !important; } }';

			document.body.appendChild( style_zap );

			var e = document.getElementById( 'wpcom_request_access_iframe' );
			e.parentNode.removeChild( e );

			document.cookie = 'wordpress_com_login_access=denied; path=/; max-age=31536000';

			return;
		}

		// Safari ITP
		if ( data.msg === 'REQUEST_ACCESS' ) {
			console.log( 'request access: safari' );

			// Check ITP iframe enable/disable knob
			if ( wpcom_remote_login_extra_auth !== 'safari_itp_iframe' ) {
				return;
			}

			// If we are in a "private window" there is no ITP.
			var private_window = false;
			try {
				var opendb = window.openDatabase( null, null, null, null );
			} catch( e ) {
				private_window = true;
			}

			if ( private_window ) {
				console.log( 'private window' );
				return;
			}

			var iframe = document.createElement( 'iframe' );
			iframe.id = 'wpcom_request_access_iframe';
			iframe.setAttribute( 'scrolling', 'no' );
			iframe.setAttribute( 'sandbox', 'allow-storage-access-by-user-activation allow-scripts allow-same-origin allow-top-navigation-by-user-activation' );

			var css = 'html { -webkit-transition: margin-top 1s; transition: margin-top 1s; } /* 9001 */ html { margin-top: 46px !important; } * html body { margin-top: 46px !important; } @media screen and ( max-width: 660px ) { html { margin-top: 71px !important; } * html body { margin-top: 71px !important; } #wpcom_request_access_iframe { display: block; height: 71px !important; } } #wpcom_request_access_iframe { border: 0px; height: 46px; position: fixed; top: 0; left: 0; width: 100%; min-width: 100%; z-index: 99999; background: #23282d; } ';

			document.body.appendChild( iframe );
		}

			wpcom_remote_login_final_cleanup();
		}
	}

<p><strong>Black Box Voting, founded in 2003, performs nonpartisan investigative reporting on elections in an attempt to stop vote rigging.</strong></p>
<p>You may be wondering what the term “black box” means. A “black box” system is non-transparent; its functions are hidden from the public. Elections, of course, should not be black box systems.</p>
<p>Author Bev Harris became known for groundbreaking work on electronic voting machines, which can remove transparency of the vote count.</p>
<p>With voting machines, all political power can be converted to the hands of a few anonymous subcontractors:</p>

Demonstration uses a real voting system and real vote databases and takes place in seconds across multiple jurisdictions. Over 5000 subcontractors and middlemen have the access to perform this for any or all clients. It can give contract signing authority to whoever the user chooses. All political power can be converted to the hands of a few anonymous subcontractors. It’s a product. It’s scaleable. It learns its environment and can adjust to any political environment, any demographic. It runs silently, invisibly, and can produce plausible results that really pass for the real thing.</p>

<p>An intriguing Ministry of Finance (<a href="http://old.minfin.ru/en/"><span style="color:#0066cc;">MoF</span></a>) report circulating in the Kremlin today says that <strong>elite Western bankers were “<i>stunned/bewildered</i>” a few hours ago after the Bank For International Settlements (<a href="http://www.bis.org/about/"><span style="color:#0066cc;">BIS</span></a>) registered a $1.8 billion transfer from the Clinton Foundation (<a href="https://www.clintonfoundation.org/"><span style="color:#0066cc;">CF</span></a>) to the Qatar Central Bank (<a href="http://www.qcb.gov.qa/English/Pages/Default.aspx"><span style="color:#0066cc;">QCB</span></a>) through the “<i>facilitation/abetment</i>” of JP Morgan Chase &amp; Company (<a href="https://www.jpmorganchase.com/"><span style="color:#0066cc;">JPM</span></a>)—and for reasons yet to be firmly established.</strong></p>
<p>According to this report, the Bank for International Settlements is the <a href="http://www.bis.org/about/"><span style="color:#0066cc;">world’s oldest international financial organization and acts as a prime counterparty for central banks in their financial transactions</span></a>; the Qatar Central Bank is the <a href="http://www.mofa.gov.qa/en/Qatar/Pages/FinancialAndBankingSystem.aspx"><span style="color:#0066cc;">bank of that Gulf State nations government and their “<i>bank of banks</i></span></a>”; JP Morgan Chase &amp; Company is <a href="https://www.incomeinvestors.com/jpmorgan-chase-co-jpm-stock-going-crazy-today/6754/"><span style="color:#0066cc;">the United States largest “<i>megabank</i></span><span style="color:#0066cc;">”</span></a>; and the Clinton Foundation <a href="http://insider.foxnews.com/2016/08/12/judge-jeanine-goes-clinton-foundation-money-laundering-op-not-charity"><span style="color:#0066cc;">is an international criminal money laundering organization</span></a> whose <a href="https://www.lewrockwell.com/2016/10/roger-stone/russian-mafia-money-launderingthe-clinton-foundation-john-podesta/"><span style="color:#0066cc;">clients include the Russian mafia</span></a>.</p>

</h1>
</header><!-- .entry-header -->
<div class="entry-content">
<p> </p>
<p><span class="embed-youtube" style="text-align:center; display: block;"><iframe allowfullscreen="true" class="youtube-player" height="334" sandbox="allow-scripts allow-same-origin allow-popups allow-presentation" src="https://www.youtube.com/embed/9q8gmMr-rCE?version=3&amp;rel=1&amp;showsearch=0&amp;showinfo=1&amp;iv_load_policy=1&amp;fs=1&amp;hl=en&amp;autohide=2&amp;wmode=transparent" style="border:0;" width="593"></iframe></span></p>
<p><strong><a href="https://www.youtube.com/channel/UChZRoC9bMegevAxFmee1oSA">By Bill Still of the STILL REPORT</a></strong></p>
<p>According to a Russian Foreign Intelligence non-classified report, reported on by Russia Today, a U.S. Admiral has been fired for questioning why the top Navy security and intelligence team was dispatched from Camp David to Dubai an what he termed an:<br/>
“… Obama house hunting mission.”<br/>
Apparently, President Obama is in the process of purchasing a $4.9 million dollar seaside villa in Dubai for his retirement. Reportedly a deposit was made on the property last week by the Podesta Group, a global public relations firm.<br/>
Interestingly, the purchase is being handled by the global law firm DLA Piper, said to be the representatives of the Kingdom of Saudi Arabia.</p>
<p><strong>On Friday, Jan. 8, Rear Admiral Rick Williams sent an email asking why the top US Navy security team was being tasked towards checking out the security implications of the new home.</strong><br/>
<strong>Within 18 hours, Adm. Williams’ boss, Vice Admiral Nora Tyson – acting on direct orders from President Obama – fired Admiral Williams</strong>.<br/>
The justification?<br/>
“Loss of confidence in his ability to command [because of] allegations of misuse of government computer equipment.”<br/>
Suddenly, anonymous reports began to circulate in the U.S. that Adm. Williams had been viewing porn on his govt. computer.<br/>
Having worked in the Pentagon environment, I can state for a fact that this is impossible. These sites have been blocked on govt computers for years and anyone who tries it, gets the infamous red screen of death that can only be unfrozen by the Help Desk – a very embarrassing proposition.<br/>
In other words, this is a lie! It’s a flimsy cover story and just one more example of Obama’s ruthless cleansing of any military officer that dares to offend him.</p>

<h1 class="entry-title">
<a href="https://counterpsyops.com/2016/10/26/did-the-clintons-commit-murder/" rel="bookmark">Did the Clintons Commit Murder?</a>
</h1>
</header><!-- .entry-header -->
<div class="entry-content">
<p><span class="embed-youtube" style="text-align:center; display: block;"><iframe allowfullscreen="true" class="youtube-player" height="334" sandbox="allow-scripts allow-same-origin allow-popups allow-presentation" src="https://www.youtube.com/embed/mX7bhd51B4M?version=3&amp;rel=1&amp;showsearch=0&amp;showinfo=1&amp;iv_load_policy=1&amp;fs=1&amp;hl=en&amp;autohide=2&amp;wmode=transparent" style="border:0;" width="593"></iframe></span></p>
<p><strong>By <a href="https://www.youtube.com/channel/UC4dD7XXdwPIrTLStHwfsJOQ">World 5 List</a></strong></p>
<p><strong>More than 12 plane crashes. 13 murders. These aren’t just any deaths. These are deaths with a connection to Bill or Hillary Clinton. </strong></p>
<p>Let’s start with <strong>Ron Brown</strong></p>
<p>There is a reason Ron Brown is the first one on this list. <strong>He was the first African American to become the US Secretary of Commerce</strong> and he served during the first term of Bill Clinton’s presidency. So how exacty did he die and why are people so suspicious of who is behind his death?</p>
<p>Brown was killed during a plane crash near Dubrovnik in Croatia, in 1996. Here’s why people think Bill Clinton was involved in the murder of Ron Brown:</p>
<p><strong>Bill Shelton and Kathy Ferguson</strong></p>
<p><strong>Kathy Ferguson was a witness during Clinton’s Paula Jones lawsuit and she was found dead in Shelton’s apartment, with a gunshot wound in her head.</strong></p>
<p>And Shelton? A month later, her fiancee killed himself at her grave site. Okay, sounds like your typical heartbreak Romeo and Juliet material, right?</p>
<p>And we’re moving right along to Barbara Wise.</p>
<p>Wise’s death is by far the most mysterious out of all on our list. The Commerce Department employee apparently worked closely with Ron Brown and John Huang.</p>
<p>Someone must’ve really had deep hatred for Wise because the woman was found dead and locked in her own office. She was bruised from head to waist and nobody ever discovered the reason why, however, plenty of speculations have occured and many do think she was tortured for hours. Does The Commerce Department not have any cameras around their building?</p>
<p>And here comes another shot in the head victim.</p>
<p><strong>Ed Wiley was involved in plenty of Clinton’s fundraising events. Cause of death? You guessed it – , with a perfect shot in the head.</strong> But wait, it gets even weirder once we give you some background info.</p>
<p>In 1993, Wiley’s wife, Kathleen Wiley, had a job-seeking meeting with Clinton as the couple faced plenty of financial difficulties.</p>
<p>Kathleen is still very anti-clinton and has already expressed her desires for Trump to win the upcoming election.</p>
<p>You’ve probably heard of <strong>Stanley Heard and Steve Dickson</strong>, right?</p>
<p>Dickson was a counsel to Heard and Heard was a counsel to Clinton’s mother, stepfather and brother. Anything else these two have in common? Both died in a plane crash. On the exact same plane.</p>
<p>And there’s the one that wasn’t shot in the head.</p>
<p><strong>The murder of Seth Rich</strong>, <strong>a Democratic National Committee staffer, killed in July 2016. The police found his body on the floor and was quick to rule it as an attempted robbery. Want to hear something odd? All his personal items, wallet and phone included, were still on him. </strong></p>
<p>What’s so special about this guy, behind his obvious last name, and why would he even be a target? People suspect he could’ve been involved in the large DNC email Wikileaks scandal and persuaded by Sanders.</p>
<p>Have you ever heard people say “if I’m ever found dead, it was murder” in movies?</p>
<p>Well, it happened in real life too. <strong>Victor Thorn is the next alleged murder Clinton murder victim. Thorn wrote several books exposin both of the Clintons – he allegedly killed himself on his birthday. Unlike most of the victims, he shot himself on top of a mountain.</strong> Was it a headshot? We don’t know. But if it was, we know where this story is going.</p>
<p><strong>Thorn told the Russel Scott Show host “Russel, if I’m ever found dead, it was murder. I would never kill myself.”</strong></p>
<p>Are you sure of that, Mr Thorn?</p>
<p>Alright, this is getting interesting. Who’s gonna be next?</p>
<p><strong>Jerry Parks. Parks was Clinton’s head of gubernatorial security team in Little Rock.</strong> <strong>He was shot seven times at a deserted intersection while driving through Little Rock.</strong></p>
<p><strong>Kevin Ives and Donald Henry</strong></p>
<p>Why people think their deaths weren’t accidental?</p>
<p><strong>Autopsies revealed that kevin died from a crushed skull before he was ever placed on the tracks while Donald was stabbed in the back – and not metaphorically.</strong></p>
<p>And we’re not quite done with their story. <strong>Kevin and Donald are connected to the death of John Hillyer. </strong></p>
<p><strong>The NBC cameraman was working on several cases that could easily expose the Clintons – he was also involved in exposing the Arkansas drug smuggling operation and worked on the case with journalist Pat Matrisciana</strong>. The two also became friends of Gary Parks, the son of Jerry Parks who we mentioned earlier.</p>
